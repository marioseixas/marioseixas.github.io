```
#!/bin/bash

# Defina o diretório raiz da pesquisa e o diretório de destino para os arquivos de texto
SEARCH_DIR="/caminho/para/o/diretorio"
DEST_DIR="/root"
FINAL_FILE="${DEST_DIR}/final_concatenated_file.txt"

# Limpa o arquivo final se ele já existir
> "$FINAL_FILE"

# Encontra todos os arquivos no diretório e subdiretórios
find "$SEARCH_DIR" -type f | while read file; do
    # Gera um nome de arquivo de texto baseado no nome do arquivo original
    TEXT_FILE_NAME=$(echo "$file" | sed 's/[^a-zA-Z0-9]/_/g')
    TEXT_FILE_PATH="${DEST_DIR}/${TEXT_FILE_NAME}.txt"
    
    # Extrai o conteúdo para o arquivo de texto no diretório /root
    cat "$file" > "$TEXT_FILE_PATH"
    
    # Adiciona o conteúdo do arquivo de texto ao arquivo final, com separador
    echo '***' >> "$FINAL_FILE"
    cat "$TEXT_FILE_PATH" >> "$FINAL_FILE"
done

# Informa ao usuário que o processo foi concluído
echo "Processo concluído. Todos os arquivos foram processados e concatenados em ${FINAL_FILE}."
```
